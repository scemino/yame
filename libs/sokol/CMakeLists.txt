if (FIPS_WINDOWS)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
endif()

# the general sokol implementations library (compiled as C, C++ or ObjC depending on platform)
fips_begin_lib(sokol)
    fips_files(clock.c clock.h fs.c fs.h gfx.c gfx.h)
    sokol_shader(shaders.glsl ${slang})
    if (FIPS_OSX)
        fips_files(sokol.m)
        fips_frameworks_osx(Foundation)
        if (FIPS_IOS)
            fips_frameworks_osx(UIKit Metal MetalKit AudioToolbox AVFoundation)
        else()
            fips_frameworks_osx(Cocoa QuartzCore Metal MetalKit AudioToolbox)
        endif()
	if (FIPS_MSVC)
            target_compile_options(sokol PRIVATE /W4)
	endif()
    else()
        fips_files(sokol.c)
        if (FIPS_ANDROID)
            fips_libs(GLESv3 EGL OpenSLES android log)
        elseif (FIPS_LINUX)
            fips_libs(X11 Xcursor Xi GL m dl asound)
        endif()
    endif()
fips_end_lib()

# optional UI library (using Dear ImGui)
fips_begin_lib(ui)
    fips_files(ui.cc ui.h)
    fips_deps(imgui-docking)
fips_end_lib()
